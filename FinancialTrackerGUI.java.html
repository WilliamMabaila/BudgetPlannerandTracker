48       // Initializes the financial tracker by prompting the user to enter the budget name and total budget amount.
49       private void initializeFinancialTracker() {
50           // Prompt for budget name and total budget to start a new financial tracker or load an existing one.
51           String budgetName = JOptionPane.showInputDialog("Enter the name for your budget:");
52           double totalBudget = Double.parseDouble(JOptionPane.showInputDialog("Enter the total budget amount:"));
53           financialTracker = new FinancialTracker(budgetName, totalBudget);
54       }

78       //  Sets up the display options panel with a combo box allowing the user to choose between viewing all transactions,
79       //  only incomes, or only expenses.
80       private void setupDisplayOptions() {
81           JPanel displayOptionsPanel = new JPanel(new FlowLayout(FlowLayout.LEFT));
82           JLabel displayOptionsLabel = new JLabel("Display: ");
83           displayOptionsComboBox = new JComboBox<>(new String[]{"All", "Incomes", "Expenses"});
84           displayOptionsComboBox.addActionListener(this::displayOptionsAction);
85   
86           displayOptionsPanel.add(displayOptionsLabel);
87           displayOptionsPanel.add(displayOptionsComboBox);
88           frame.add(displayOptionsPanel, BorderLayout.NORTH);
89       }

92       // Handles the action of changing the display options to show either all transactions, only incomes,
93       // or only expenses.
94       private void displayOptionsAction(ActionEvent e) {
95           String selectedOption = (String) displayOptionsComboBox.getSelectedItem();
96           switch (selectedOption) {
97               case "All":
98                   updateTable(); // Update the table with both incomes and expenses
99                   break;
100              case "Incomes":
101                  updateTableWithIncomes(); // Update the table with incomes only
102                  break;
103              case "Expenses":
104                  updateTableWithExpenses(); // Update the table with expenses only
105                  break;
106          }
107      }

142       // Opens a dialog for the user to enter details about a new income, which is then added to the financial tracker
143       // and displayed in the table.
144      private void addIncomeAction(ActionEvent e) {
145          JPanel panel = new JPanel(new GridLayout(0, 2));
146          sourceField = new JTextField();
147          amountField = new JTextField();
148          frequencyField = new JTextField();
149  
150          panel.add(new JLabel("Source:"));
151          panel.add(sourceField);
152          panel.add(new JLabel("Amount:"));
153          panel.add(amountField);
154          panel.add(new JLabel("Frequency:"));
155          panel.add(frequencyField);
156  
157          int result = JOptionPane.showConfirmDialog(frame, panel, "Add Income",
158                  JOptionPane.OK_CANCEL_OPTION, JOptionPane.PLAIN_MESSAGE);
159          if (result == JOptionPane.OK_OPTION) {
160              String source = sourceField.getText();
161              double amount = Double.parseDouble(amountField.getText());
162              String frequency = frequencyField.getText();
163  
164              Income income = new Income(amount, source, frequency);
165              financialTracker.addIncome(income);
166              tableModel.addRow(new Object[]{"Income", source, amount, frequency});
167  
168              displayIncomeGif("/incomeadded.gif");
169          }
170      }

185      // Opens a dialog for the user to enter details about a new expense, which is then added to the financial tracker
186      // and displayed in the table. The dialog includes fields for the expense category, amount, description, and
187      // whether it is recurring.
188      private void addExpenseAction(ActionEvent e) {
189          JPanel panel = createExpenseDialogPanel();
190  
191          int result = JOptionPane.showConfirmDialog(frame, panel, "Add Expense",
192                  JOptionPane.OK_CANCEL_OPTION, JOptionPane.PLAIN_MESSAGE);
193          if (result == JOptionPane.OK_OPTION) {
194              String category = categoryField.getText();
195              double amount = Double.parseDouble(amountField.getText());
196              String description = descriptionField.getText();
197              boolean isRecurring = isRecurringBox.isSelected();
198  
199              Expense expense = new Expense(amount, category, description, isRecurring);
200              financialTracker.addExpense(expense);
201              tableModel.addRow(new Object[]{"Expense", category, amount, isRecurring ? "Yes" : "No"});
202  
203              displayExpenseGif("/expenseadded.gif");
204          }
205      }

241      // Saves the current state of the financial tracker to a JSON file. Displays a confirmation message on successful
242       // save or an error message if the file cannot be written.
243      private void saveFinancialTracker() {
244          try {
245              jsonWriter.open();
246              jsonWriter.write(financialTracker);
247              jsonWriter.close();
248              JOptionPane.showMessageDialog(frame, "Financial tracker saved successfully.", "Save",
249                      JOptionPane.INFORMATION_MESSAGE);
250          } catch (FileNotFoundException ex) {
251              JOptionPane.showMessageDialog(frame, "Unable to save financial tracker.", "Error",
252                      JOptionPane.ERROR_MESSAGE);
253          }
254      }

257       // Loads the financial tracker data from a JSON file. Updates the table to display the loaded incomes and expenses.
258       // Displays a confirmation message on successful load or an error message if the file cannot be read.
259      private void loadFinancialTracker() {
260          try {
261              financialTracker = jsonReader.read();
262              JOptionPane.showMessageDialog(frame, "Financial tracker loaded successfully.",
263                      "Load", JOptionPane.INFORMATION_MESSAGE);
264              updateTable();
265          } catch (IOException ex) {
266              JOptionPane.showMessageDialog(frame, "Unable to load financial tracker.",
267                      "Error", JOptionPane.ERROR_MESSAGE);
268          }
269      }

272       // Updates the table to display all transactions (both incomes and expenses) in the financial tracker. Clears
273       // the table before adding the new data.
274      private void updateTable() {
275          clearTable();
276          for (Income income : financialTracker.returnIncome()) {
277              tableModel.addRow(new Object[]{"Income", income.getSource(), income.getAmount(), income.getFrequency()});
278          }
279          for (Expense expense : financialTracker.returnExpense()) {
280              tableModel.addRow(new Object[]{"Expense", expense.getCategory(), expense.getAmount(),
281                      expense.isRecurring() ? "Yes" : "No"});
282          }
283      }
284  
285  
286       // Updates the table to display only income transactions in the financial tracker. Clears the table before
287       // adding the new data.
288      private void updateTableWithIncomes() {
289          clearTable();
290          for (Income income : financialTracker.returnIncome()) {
291              tableModel.addRow(new Object[]{"Income", income.getSource(), income.getAmount(), income.getFrequency()});
292          }
293      }
294  
295  
296       // Updates the table to display only expense transactions in the financial tracker. Clears the table before
297       // adding the new data.
298      private void updateTableWithExpenses() {
299          clearTable();
300          for (Expense expense : financialTracker.returnExpense()) {
301              tableModel.addRow(new Object[]{"Expense", expense.getCategory(), expense.getAmount(),
302                      expense.isRecurring() ? "Yes" : "No"});
303          }
304      }

307       // Clears all data from the table, removing all rows.
308      private void clearTable() {
309          tableModel.setRowCount(0);
310      }
